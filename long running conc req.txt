SELECT DISTINCT
    c.user_concurrent_program_name,
    round( ( (SYSDATE - a.actual_start_date) * 24 * 60 * 60 / 60),2) AS process_time,
    a.request_id,
    e.sid,
    e.inst_id,
    e.sql_id,
    f.plan_hash_value,
    a.outfile_node_name,
    a.parent_request_id,
    a.request_date,
    a.actual_start_date,
    a.actual_completion_date,
    ( a.actual_completion_date - a.request_date ) * 24 * 60 * 60 AS end_to_end,
    ( a.actual_start_date - a.request_date ) * 24 * 60 * 60 AS lag_time,
    d.user_name,
    a.phase_code,
    a.status_code,
    a.argument_text,
    a.priority,
    e.sid,
    e.serial#,
    e.status,
    e.action,
    e.module
FROM
    apps.fnd_concurrent_requests a,
    apps.fnd_concurrent_programs b,
    apps.fnd_concurrent_programs_tl c,
    apps.fnd_user d,
    gv$session e,
    gv$sqlarea f
WHERE
    a.concurrent_program_id = b.concurrent_program_id
    AND   b.concurrent_program_id = c.concurrent_program_id
    AND   a.requested_by = d.user_id
    AND   e.audsid = a.ORACLE_SESSION_ID
    AND   f.sql_id = e.sql_id
    AND   status_code = 'R'
ORDER BY
    process_time DESC;
